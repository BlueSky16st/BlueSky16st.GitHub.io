### 白米饭(1 人份)

> 状态: 尝试中

```TypeScript
import { 量杯, 电饭煲, 碗 } from '厨具';

interface Ingredient {
  米: 量杯;
  // 单位: ML
  水: number;
}

//TODO: 尝试中
const 材料: Ingredient = {
  米: new 量杯(刻度: 3/4),
  水: 150,
}
const 煮饭信息: TimeInfo {
  时长: 40,
  是否需要保温: true
}

const start = () => {
  const 电饭煲 = new 电饭煲();
  电饭煲.内胆.push(材料.米);
  电饭煲.内胆.淘米(次数: randomInt(2, 3));
  电饭煲.内胆.push(材料.水);

  电饭煲.插电();
  电饭煲.setTime(time: 煮饭信息.时间, isHeatPreservation: 煮饭信息.是否需要保温);
  电饭煲.start().then(() => {
    const 碗 = new 碗();
    碗.push(电饭煲.内胆.pop());
    console.log("煮饭完成了!");
  });
}

start();
```
